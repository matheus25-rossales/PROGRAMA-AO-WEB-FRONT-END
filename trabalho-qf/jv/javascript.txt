
document.querySelector("#enviar").addEventListener("click", () => {
  const confirmado = document.querySelector("input[name='q1']:checked");

  if (confirmado && confirmado.value === "B") {
    // cria ou atualiza uma mensagem de feedback
    let feedback = document.querySelector("#p1 .feedback");
    if (!feedback) {
      feedback = document.createElement("p");
      feedback.classList.add("feedback");
      document.querySelector("#p1").appendChild(feedback);
    }
     feedback.innerHTML = "<br> ✅ Resposta correta! ";
  } else {let feedback = document.querySelector("#p1 .feedback");
    if (!feedback) {
      feedback = document.createElement("p");
      feedback.classList.add("feedback");
      document.querySelector("#p1").appendChild(feedback);
    }
    feedback.innerHTML = "<br> ❌ Resposta incorreta  ou nenhuma opção marcada.";
  }
});
